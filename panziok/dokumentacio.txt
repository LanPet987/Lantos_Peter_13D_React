MI válasz:
SELECT 
    szobak.sznev AS "Szoba", 
    foglalasok.fo AS "Vendegek", 
    (foglalasok.tav - foglalasok.erk) AS "Éjszakák"
FROM 
    szobak
INNER JOIN 
    foglalasok 
ON 
    szobak.szazon = foglalasok.szoba
ORDER BY 
    foglalasok.szoba DESC;
